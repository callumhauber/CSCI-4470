## Part 2
### Makefile
```
vpath %.h headers
vpath %.c source

all: static_block dynamic_block

static_block: program.o libblock.a
	cc program.o libblock.a -o static_block -Wl,-rpath .

dynamic_block: program.o libblock.so
	cc program.o libblock.so -o dynamic_block -Wl,-rpath .

program.o: program.c
	cc -c $< -o $@

libblock.so: block.o
	cc -shared -o libblock.so block.o

libblock.a: block.o
	ar qc libblock.a block.o

block.o: block.c
	cc -fPIC -c $< -o $@

block.o: block.h

clean:
	rm -f dynamic_block static_block *.a *.so *.o all
```

### CMake
CMakeLists.txt
```
cmake_minimum_required(VERSION 3.10)

# set the project name and version
project(Block)

# add the executables
add_executable(cmake_static_block program.c)

# add the executables
add_executable(cmake_dynamic_block program.c)

set(SOURCE_PATH "./source")
set(HEADER_PATH "./headers")

# make libraries
add_library(cmake_libblock_static STATIC ${SOURCE_PATH}/block.c ${HEADER_PATH}/block.h)
add_library(cmake_libblock_shared SHARED ${SOURCE_PATH}/block.c ${HEADER_PATH}/block.h)

target_link_libraries(cmake_static_block cmake_libblock_static)
target_link_libraries(cmake_dynamic_block cmake_libblock_shared)
```
